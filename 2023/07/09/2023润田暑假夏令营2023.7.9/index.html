sed -i -e "s|#baseurl=http://mirror.centos.org|baseurl=http://vault.centos.org|g" /etc/yum.repos.d/CentOS-*

配置WAF环境
1.上传文件 image.tar.gz(运行的底层逻辑镜像）
2.cat image.tar.gz | gzip -d | docker laod //加载镜像
3.docker ps -a//查看有无容器在运行
4.mkdir safeline//创建文件夹
5.cd safeline//查看文件夹
6.上传
（ll -a//查看文件夹文件包含隐藏文件）

echo “SAFELINE_DIR=$safeline_path” >> .env
echo “IMAGE_TAG=latest” >> .env
echo “MGT_PORT=9443” >> .env
echo “POSTGRES_PASSWORD=$(LC_ALL=C tr -dc A-Za-z0-9 </dev/urandom | head -c32)” >> .env
echo “REDIS_PASSWORD=$(LC_ALL=C tr -dc A-Za-z0-9 </dev/urandom | head -c 32)”>> .env
echo “SUBNET_PREFIX=169.254.0” >> .env
